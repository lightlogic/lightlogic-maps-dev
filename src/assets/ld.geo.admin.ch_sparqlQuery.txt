PREFIX schema: <http://schema.org/>
PREFIX geo: <http://www.opengis.net/ont/geosparql#>
PREFIX gn: <http://www.geonames.org/ontology#>

SELECT ?Commune ?Name ?WKT
WHERE {
?Commune gn:featureCode gn:A.ADM3 .
?Commune schema:name "Mont-Vully" .
?Commune geo:defaultGeometry ?Geometry .
?Geometry geo:asWKT ?WKT .
?Commune <http://purl.org/dc/terms/issued> ?Date .
FILTER (?Date = "2020-01-01"^^xsd:date)
}

PREFIX ld: <http://linkeddata.ru/>
PREFIX dcterms: <http://purl.org/dc/terms/>
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#>
PREFIX gn: <http://www.geonames.org/ontology#>
PREFIX schema: <http://schema.org/>
PREFIX dcterm: <http://purl.org/dc/terms/>
PREFIX ldadm: <https://ld.geo.admin.ch/def/>

SELECT ?communeUri ?communeName ?cantonUri ?cantonName ?bfsNum
WHERE {
  ?communeUri gn:featureCode <http://www.geonames.org/ontology#A.ADM3>;
    schema:name "Ins";
              schema:name ?communeName;
	ldadm:bfsNumber ?bfsNum ;
    gn:parentADM1 ?cantonUri ;
	<http://purl.org/dc/terms/issued> ?Date .
 ?cantonUri schema:name ?cantonName.
  FILTER (?Date = "2020-01-01"^^xsd:date)
}
LIMIT 100
