PREFIX schema: <http://schema.org/>
PREFIX geo: <http://www.opengis.net/ont/geosparql#>
PREFIX gn: <http://www.geonames.org/ontology#>

SELECT ?Commune ?Name ?WKT
WHERE {
?Commune gn:featureCode gn:A.ADM3 .
?Commune schema:name "Mont-Vully" .
?Commune geo:defaultGeometry ?Geometry .
?Geometry geo:asWKT ?WKT .
?Commune <http://purl.org/dc/terms/issued> ?Date .
FILTER (?Date = "2020-01-01"^^xsd:date)
}
